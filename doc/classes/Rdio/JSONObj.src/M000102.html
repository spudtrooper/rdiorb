    def method_missing(method,args={})
      meth = method.to_s
      res = @json[meth]
      #
      # Maybe this should be a number
      #
      if meth =~ /_count$/ or meth =~ /^num_/
        begin
          return res.to_i
        rescue Exception => e
          STDERR.puts "#{meth} (err) #{e}"
        end
      end
      return res
    end
