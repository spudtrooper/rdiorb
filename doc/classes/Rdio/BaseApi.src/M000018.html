    def call(method,args)
      if Rdio::log_methods
        Rdio::log "Called method: #{method}"
      end
      new_args = {}
      new_args['method'] = method
      args.each do |k,v|
        new_args[k] = BaseApi.key(v).to_s
      end
      url = 'http://api.rdio.com/1/'
      resp,data = access_token.post url,new_args
      return data
    end
